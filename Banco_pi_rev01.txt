CREATE TABLE USUARIO (
    ID_USU      INT          PRIMARY KEY AUTO_INCREMENT,
    NOME        VARCHAR(150) NOT NULL,
    EMAIL       VARCHAR(150) NOT NULL UNIQUE,
    SENHA       VARCHAR(255) NOT NULL,
    PERFIL      ENUM('administrador', 'operador', 'visualizador')
                NOT NULL DEFAULT 'visualizador',
    ATIVO       BOOLEAN      NOT NULL DEFAULT TRUE,
    DT_CADASTRO TIMESTAMP    DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE MAQUINA (
    ID_MAQUINA    INT          PRIMARY KEY AUTO_INCREMENT,
    LINHA_PROD    VARCHAR(100),
    FABRICANTE    VARCHAR(100),
    MODELO        VARCHAR(100),
    NUM_SERIE     VARCHAR(100) UNIQUE,
    DT_INSTALACAO DATE
);

CREATE TABLE STATUS_MAQUINA (
    ID_STATUS  INT          PRIMARY KEY AUTO_INCREMENT,
    ID_MAQUINA INT          NOT NULL,
    STATUS     ENUM('ativa', 'manutencao', 'inativa')
               NOT NULL DEFAULT 'ativa',
    MOTIVO     VARCHAR(255),
    DT_HORA    TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (ID_MAQUINA) REFERENCES MAQUINA(ID_MAQUINA)
);

CREATE TABLE SENSOR (
    ID_SENSOR   INT            PRIMARY KEY AUTO_INCREMENT,
    ID_MAQUINA  INT            NOT NULL,
    TIPO        VARCHAR(80)    NOT NULL,
    UNIDADE     VARCHAR(20)    NOT NULL,
    DESCRICAO   VARCHAR(150),
    TEMP_LIMITE DECIMAL(8, 2),
    ATIVO       BOOLEAN        NOT NULL DEFAULT TRUE,
    FOREIGN KEY (ID_MAQUINA) REFERENCES MAQUINA(ID_MAQUINA)
);

CREATE TABLE HISTORICO_ALTERACAO (
    ID_HISTORICO   INT           PRIMARY KEY AUTO_INCREMENT,
    ID_USU         INT           NOT NULL,
    TABELA         VARCHAR(50)   NOT NULL,        -- 'MAQUINA' ou 'SENSOR'
    ID_REGISTRO    INT           NOT NULL,        -- ID do registro alterado
    CAMPO          VARCHAR(100)  NOT NULL,        -- coluna que foi alterada
    VALOR_ANTERIOR VARCHAR(255),                  -- valor antes (nullable: novo cadastro)
    VALOR_NOVO     VARCHAR(255)  NOT NULL,        -- valor depois
    DT_ALTERACAO   TIMESTAMP     DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (ID_USU) REFERENCES USUARIO(ID_USU)
);

CREATE TABLE LEITURA_SENSOR (
    ID_LEITURA_SENSOR INT            PRIMARY KEY AUTO_INCREMENT,
    ID_MAQUINA        INT            NOT NULL,
    ID_SENSOR         INT            NOT NULL,
    VALOR             DECIMAL(12, 4) NOT NULL,
    DATA_HORA         TIMESTAMP      DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (ID_MAQUINA) REFERENCES MAQUINA(ID_MAQUINA),
    FOREIGN KEY (ID_SENSOR)  REFERENCES SENSOR(ID_SENSOR)
);


CREATE TABLE ALERTA (
    ID_ALERTA         INT            PRIMARY KEY AUTO_INCREMENT,
    ID_LEITURA_SENSOR INT            NOT NULL,
    ID_MAQUINA        INT            NOT NULL,
    TIPO              ENUM('temp_alta', 'vibracao', 'pressao', 'outro')
                      NOT NULL DEFAULT 'temp_alta',
    VALOR_DETECTADO   DECIMAL(8, 2)  NOT NULL,
    VALOR_LIMITE      DECIMAL(8, 2)  NOT NULL,    -- snapshot de SENSOR.TEMP_LIMITE
    CRITICIDADE       ENUM('baixa', 'media', 'alta')
                      NOT NULL DEFAULT 'media',
    VISUALIZADO       BOOLEAN        NOT NULL DEFAULT FALSE,
    DATA_HORA         TIMESTAMP      DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (ID_LEITURA_SENSOR) REFERENCES LEITURA_SENSOR(ID_LEITURA_SENSOR),
    FOREIGN KEY (ID_MAQUINA)        REFERENCES MAQUINA(ID_MAQUINA)
);


CREATE TABLE ORDEM_PRODUCAO (
    ID_ORDEM          INT            PRIMARY KEY AUTO_INCREMENT,
    ID_ALERTA         INT,
    ID_MANUT_ANTERIOR INT,
    ID_MAQUINA        INT            NOT NULL,
    ID_USU_ABERTURA   INT,
    TIPO              ENUM('emergencial', 'preditiva', 'preventiva', 'corretiva')
                      NOT NULL,
    LINHA_PROD        VARCHAR(100)   NOT NULL,
    TEMPERATURA       DECIMAL(8, 2),
    STATUS            ENUM('aberta', 'em_andamento', 'concluida', 'cancelada')
                      NOT NULL DEFAULT 'aberta',
    PRIORIDADE        ENUM('baixa', 'media', 'alta', 'critica')
                      NOT NULL DEFAULT 'media',
    OBSERVACAO        TEXT,
    DT_ABERTURA       TIMESTAMP      DEFAULT CURRENT_TIMESTAMP,
    DT_PREVISTA       TIMESTAMP NULL DEFAULT NULL,
    DT_CONCLUSAO      TIMESTAMP NULL DEFAULT NULL,
    FOREIGN KEY (ID_ALERTA)       REFERENCES ALERTA(ID_ALERTA),
    FOREIGN KEY (ID_MAQUINA)      REFERENCES MAQUINA(ID_MAQUINA),
    FOREIGN KEY (ID_USU_ABERTURA) REFERENCES USUARIO(ID_USU)
);


CREATE TABLE MANUTENCAO (
    ID_MANUTENCAO    INT       PRIMARY KEY AUTO_INCREMENT,
    ID_ORDEM         INT       NOT NULL UNIQUE,
    ID_MAQUINA       INT       NOT NULL,
    DESCRICAO        TEXT,
    SOLUCAO          TEXT,
    PRAZO_PROX_MANUT INT,
    DT_REALIZADA    TIMESTAMP NOT NULL,
    FOREIGN KEY (ID_ORDEM)   REFERENCES ORDEM_PRODUCAO(ID_ORDEM),
    FOREIGN KEY (ID_MAQUINA) REFERENCES MAQUINA(ID_MAQUINA)
);
